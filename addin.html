const process = async (f, t) => { elResults.innerHTML = '

Analizando viajes y eventos de ocupación...

'; currentReportData = []; const fromDate = new Date(f + "T00:00:00Z").toISOString(); const toDate = new Date(t + "T23:59:59Z").toISOString(); try { const allRules = await api.call("Get", { typeName: "Rule" }); const ocupacionRule = allRules.find(r => r.name.toLowerCase().includes("pasajero") || r.name.toLowerCase().includes("ocupaci") ); if (!ocupacionRule) { elResults.innerHTML = `

Error: No se encontró la regla de ocupación.

`; return; } const [devices, allTrips, allExceptions] = await Promise.all([ api.call("Get", { typeName: "Device" }), api.call("Get", { typeName: "Trip", search: { fromDate, toDate } }), api.call("Get", { typeName: "ExceptionEvent", search: { fromDate, toDate, ruleSearch: { id: ocupacionRule.id } } }) ]); let report = devices.map(dev => { const devTrips = allTrips.filter(tr => tr.device.id === dev.id); const devExcs = allExceptions.filter(ex => ex.device.id === dev.id); let stats = { name: dev.name, occKm: 0, occTime: 0, occTrips: 0, vacKm: 0, vacTime: 0, vacTrips: 0, waitingTime: 0 }; devTrips.forEach(tr => { const tStart = new Date(tr.start); const tStop = new Date(tr.stop); const dist = tr.distance || 0; const duration = tStop - tStart; // Nueva Lógica: ¿Hubo alguna excepción de ocupación DURANTE este viaje? const hasOccupation = devExcs.some(ex => { const exStart = new Date(ex.activeFrom); const exEnd = new Date(ex.activeTo || ex.activeFrom); // El evento de ocupación ocurre dentro del rango del viaje return (exStart <= tStop && exEnd >= tStart); }); if (hasOccupation) { stats.occKm += dist; stats.occTime += duration; stats.occTrips++; } else { if (dist > 0.1) { // Solo contar como viaje vacío si hubo movimiento real stats.vacKm += dist; stats.vacTime += duration; stats.vacTrips++; } else { stats.waitingTime += duration; } } }); return stats; }).filter(s => (s.occKm + s.vacKm) > 0 || s.occTrips > 0); report.sort((a, b) => b.occTrips - a.occTrips); currentReportData = report; render(report); } catch (err) { elResults.innerHTML = `

Error: ${err.message}

`; } };
